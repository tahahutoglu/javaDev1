import java.util.Scanner;
public class Main
{
	public static void main(String[] args) {
        int sayi=100,check;
        for(int i=2;i<sayi;i++){
            check = 0;
            for(int j=2;j<i;j++){
                if(i%j==0){
                    check++;
                    break;
                }
            }
            if(check == 0){
                System.out.print(i + " ");
            }
        }
	}
}
