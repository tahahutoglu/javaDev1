import java.util.Scanner;
public class Main
{
    public static int  ust(int alt,int ust){
        if(ust == 0)return 1;
        return alt*ust(alt,ust-1);
    }
    public static void main(String[] args) {
        Scanner scan = new Scanner(System.in);
        System.out.println("alt sayı giriniz:");
        int alt = scan.nextInt();
        System.out.println("ust sayı giriniz:");
        int ust = scan.nextInt();
        int sonuc = ust(alt,ust);
        System.out.println(sonuc);
    }
}
