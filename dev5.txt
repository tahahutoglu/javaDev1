import java.util.Scanner;
public class dev1 {
    public static void main(String[] args) {
        int cap,aci;
        double PI = 3.14;
        Scanner input = new Scanner(System.in);
        System.out.println("Çapı giriniz:");
        cap = input.nextInt();
        double alan = PI * cap * cap;
        double cevre = PI * cap * 2;
        System.out.println("Alan, Cevre = " + alan + "," + cevre);
        System.out.println("Açıyı giriniz:");
        aci = input.nextInt();
        double dilimAlan = (PI * Math.pow(cap,2) * aci)/360;
        System.out.println("Dilim alan: " + dilimAlan);
    }
}
